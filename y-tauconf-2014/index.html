Factorial

				function commonFactorial(n) {
					return n === 0 ? 1 : n * commonFactorial(n-1);
				}

Non-recursive factorial

				function almostFactorial(self) {
					return function(n) {
						return n === 0 ? 1 : n * self(n-1);
					};
				}

Notation

			function almostFactorial(self)(n) {
				return n === 0 ? 1 : n * self(n-1);
			}

Fixed point

commonFactorial
is a fixed point of 
almostFactorial
!

				almostFactorial(commonFactorial)(4) === commonFactorial(4); 
				almostFactorial(commonFactorial)(10) === commonFactorial(10);
				almostFactorial(commonFactorial)(17) === commonFactorial(17);
				...
				//But we are trying to calculate factorial without recursion!

Y comes to help!

What if there exists function 
Y
that we could pass 
almostFactorial
to that would return the fixed point of it, i.e. 
commonFactorial
function?

			Y(almostFactorial)(42) =>
			commonFactorial(42) =>
			almostFactorial(commonFactorial)(42) =>
			almostFactorial(Y(almostFactorial))(42)
			//So we do not need commonFactorial function at all!

Apply almostFactorial to itself

			function almostFactorial(self)(n) {
				return n === 0 ? 1 : n * self(self)(n-1);
			}
			almostFactorial(almostFactorial)(5); // => 120

Isolate fixed point function

			function almostFactorial(self)(n) {
				var fact = function(q)(x) {
					return n === 0 ? 1 : x * q(x-1);
				};
				return fact(self(self))(n);
			}

Extract factorial calculation

			function fact(self)(n) {
				return n === 0 ? 1 : n * self(n-1);
			}
			function almostFactorial(self)(n) {
				return fact(self(self))(n);
			}
			almostFactorial(almostFactorial)(5); // => 120

Fixed point combinator

			function fix(f) {
				var p = function(self)(n) {
					return f(self(self))(n);
				};
				return p(p);
			}
			var factorial = fix(fact);
			factorial(6); // => 720

Wrappers

Simple logging wrapper

			function logWrapper(f)(self)(n) {
				var result = f(self)(n);
				console.log('f(' + n + ') = ' + result);
				return result;
			}
			var logFactorial = fix(logWrapper(fact));
			logFactorial(3); // => 6

Heterogenous wrappers

			function argListWrapper(f)(self)(acc)(n) {
				acc.push(n);
				console.log('Arguments passed: ' + acc);
				return f(self(acc))(n);
			};
			var argListFactorial = fix(argListWrapper(fact))([]);
			argListFactorial(3); // => 6

Putting wrappers together

			var compositionFactorial = fix(
								argListWrapper(
									logWrapper(
										fact
									)
								)
							)([]);
			compositionFactorial(7); // => 5040

Trampoline

Tail call optimization

			function commonFactorial(n, r) {
				return n === 0 ? r : commonFactorial(n - 1, n*r);
			};
		
			function fact(self)(n, r) {
				return n === 0 ? r : self(n - 1, n*r);
			}
			//But tail recursion isn't supported everywhere!

Trampoline

			function fix(f)(n, r) {
				var a = null;
				for (a = function(n_, r_) { n = n_; r = r_; return a; }; 
					a != null && a instanceof Function; 
					a = f(a)(n, r));//Magic is here!
				return a;
			};
			var factorial = fix(fact);
			factorial(6, 1); // => 720

Quine

A quine is a program which prints its own listing.

				(function $(){console.log('('+$+'());')}());

Quine

It is impossible (in most programming languages) for a program to manipulate itself (i.e. its textual representation — or a representation from which its textual representation can be easily derived) directly.

So to make this possible anyway, we write the build the program from two parts, one which call the code and one which we call the data.
